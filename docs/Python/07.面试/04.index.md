    @wraps(func)
    def wrapper(*args, **kwargs):
        t1 = time.time()
        ret = func(*args, **kwargs)
        cost_time = time.time()-t1

def count_it(func):
    count = 0
    @wraps(func)
    def wrapper(*args, **kwargs):
        nonlocal count
        count += 1
        ret = func(*args, **kwargs)
        print(count)
        return ret
    return wrapper
